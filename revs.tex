(* Code 4 *)
let rev l=
  let rec rev_aux acc l=
    match l with
    |[]->acc
    |t::q->rev_aux (t::acc) q
  in rev_aux [] l

(* Code 5 *)
let rev l =
  match l with
    [] -> []
  |a::q -> let rec rev2 x y = match y with
        [] -> x
      |b::z -> rev2 (b::x) z in rev2 [] l

let rev l =
  List.fold_left (fun acc x -> x :: acc) [] l
